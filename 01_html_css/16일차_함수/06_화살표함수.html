function make1() {
            alert('실행문');
        }
        make1();

        화살표함수
        const make1 = () => {
            alert('실행문');
        }
        make1();

        function make1(x) {
            alert(`${x} 값`)
        }
        make1();
        
        // 값이 1개일 경우 () 생략 가능
        const make1 = x =>  {
            alert(`${x} 값`)
        }
        make1(100);

        function make1(x) {
            return(`${x} 값`)
        }
        let x1 = make1(100);
        alert(x1);

        const make1 = (x) => {
            return(`${x} 값`)
        }
        let x1 = make1(100);
        alert(x1);

        // 결과가 1개
        const make1 = x => `${x} 값`
        let x1 = make1(100);
        alert(x1);
